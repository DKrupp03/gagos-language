# Isso é um comentário
fufuncao nome_funcao (parametro1, parametro2) {
  sese (parametro1 > parametro2) {
    lalaco_ate (parametro1 < parametro2) {
      escrecreve ("numero = {parametro1}") # Concatena
      parametro1 = parametro1 - 1
    }
  }
  sesenao {
    lalaco_ate (parametro1 > parametro2) {
      escrecreve ("numero = {parametro1}")
      parametro1 = parametro1 + 1
    }
  }

  reretorna parametro1
}

nome_funcao (5, 1)
